pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Install htmlhint') {
            steps {
                sh 'npm install -g htmlhint'
            }
        }
        stage('Run HTML Lint') {
            steps {
                sh '''
                htmlhint ./iPortfolio-1.0.0/*.html
                '''
            }
        }
    }
}
